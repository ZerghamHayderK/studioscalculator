", unsafe_allow_html=True) # 3. Initialization (Fixes 'list' vs 'dict' error) if 'cart' not in st.session_state or isinstance(st.session_state.cart, list): st.session_state.cart = {} # We use a dictionary to allow smart merging # 4. Data Connection url = "https://docs.google.com/spreadsheets/d/1qvBKlYH7q4dXsu7tEh9OLnKSydfONNRzGa-xjUYLB0g/edit?usp=sharing" try: conn = st.connection("gsheets", type=GSheetsConnection) df = conn.read(spreadsheet=url, ttl="1m") df.columns = [c.strip() for c in df.columns] # Remove accidental spaces # Define exact column names from your data col_cat = "Product Category" col_prod = "Product" col_pkr = "Total Price (PKR)" col_sar = "Total Price (SAR)" col_aed = "Total Price (AED)" # Numeric Cleaning Utility def clean_currency(val): if isinstance(val, str): return float(val.replace('PKR','').replace('SAR','').replace('dh','').replace(',','').strip()) return float(val) # 5. Interface with st.expander("CONFIGURE YOUR PROJECT SCOPE", expanded=True): category = st.selectbox("CATEGORY", df[col_cat].unique()) sub_df = df[df[col_cat] == category] product = st.selectbox("PRODUCT SERVICE", sub_df[col_prod].unique()) units = st.number_input("UNITS REQUIRED", min_value=1, value=1) if st.button("ADD TO QUOTATION"): row = sub_df[sub_df[col_prod] == product].iloc[0] if product in st.session_state.cart: st.session_state.cart[product]['units'] += units else: st.session_state.cart[product] = { "units": units, "pkr_unit": clean_currency(row[col_pkr]), "sar_unit": clean_currency(row[col_sar]), "aed_unit": clean_currency(row[col_aed]) } st.rerun() # 6. Cart Display if st.session_state.cart: st.markdown("---") t_pkr, t_sar, t_aed = 0, 0, 0 for name, data in st.session_state.cart.items(): sub_pkr = data['units'] * data['pkr_unit'] t_pkr += sub_pkr t_sar += data['units'] * data['sar_unit'] t_aed += data['units'] * data['aed_unit'] st.write(f"**{name}** (x{data['units']}) — PKR {sub_pkr:,.0f}") # Total Valuation Card st.markdown(f"""
